    Dim RasaWrog As String
    RasaWrog = "Goblin"

    StatusWrog = "W pe³ni si³"

    ZywotnoscWrog = Losuj(4, 7)

    ReakcjaWrog = Losuj(1, 4)
    Dim AtakWrog As Integer
